struct AgsAudioSignal

struct AgsAudioSignal {
  GObject object;

  guint flags;

  GObject *devout;

  GObject *recycling;
  GObject *recall_id; // AGS_TYPE_RECALL_ID to identify the AgsAudioSignal

  guint samplerate;
  guint buffer_size;
  guint resolution;

  guint length;
  guint last_frame; // the last frame at stream_end
  guint loop_start;
  guint loop_end;

  guint delay;
  guint attack;

  guint lock_attack;
  
  GList *stream_beginning;
  GList *stream_current;
  GList *stream_end;
};
